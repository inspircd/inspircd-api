111 const std::string val = getString(key);
112 if (val.empty())
113 return def;
114
115 for (const std::pair<const char*, TReturn>& enumval : enumvals)
116 if (stdalgo::string::equalsci(val, enumval.first))
117 return enumval.second;
118
119 std::vector<const char*> enumkeys;
120 std::transform(enumvals.begin(), enumvals.end(), std::back_inserter(enumkeys), [](const std::pair<const char*, TReturn>& ev) { return ev.first; });
121 throw CoreException(val + " is an invalid value for <" + name + ":" + key + ">; acceptable values are " +
122 stdalgo::string::join(enumkeys, ' ') + ", at " + source.str());
123 }
