108 template<typename TReturn>
109 TReturn getEnum(const std::string& key, TReturn def, std::initializer_list<std::pair<const char*, TReturn>> enumvals)
110 {
111 const std::string val = getString(key);
112 if (val.empty())
113 return def;
114
115 for (const auto& [enumkey, enumval] : enumvals)
116 if (stdalgo::string::equalsci(val, enumkey))
117 return enumval;
118
119 // Unfortunately we have to iterate this twice.
120 std::string enumdef = "(unknown)";
121 std::string enumkeys;
122 for (const auto& [enumkey, enumval] : enumvals)
123 {
124 enumkeys.append(enumkey).append(", ");
125 if (enumval == def)
126 enumdef = enumkey;
127 }
128 if (!enumkeys.empty())
129 enumkeys.erase(enumkeys.length() - 2);
130
131 LogMalformed(key, val, enumdef, "not one of " + enumkeys);
132 return def;
133 }
