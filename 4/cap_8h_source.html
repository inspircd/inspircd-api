194 bool IsEnabled(User* user) const
195 {
196 if (!IsRegistered())
197 return false;
198 Ext caps = extitem->Get(user);
199 return ((caps & GetMask()) != 0);
200 }
201
206 void Set(User* user, bool val)
207 {
208 if (!IsRegistered())
209 return;
210 Ext curr = extitem->Get(user);
211 extitem->Set(user, (val ? AddToMask(curr) : DelFromMask(curr)));
212 }
213
220 void SetActive(bool activate)
221 {
222 active = activate;
223 if (manager)
224 {
225 if (activate)
226 manager->AddCap(this);
227 else
228 manager->DelCap(this);
229 }
230 }

246 bool IsRegistered() const { return (extitem != NULL); }
