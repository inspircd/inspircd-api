208template <typename T, typename Del = std::default_delete<T>>
209class SimpleExtItem
210 : public ExtensionItem
211{
212protected:
214 bool synced;
215
216public:
223 SimpleExtItem(Module* owner, const std::string& key, ExtensionType exttype, bool sync = false)
224 : ExtensionItem(owner, key, exttype)
225 , synced(sync)
226 {
227 }
228
230 void Delete(Extensible* container, void* item) override
231 {
232 Del del;
233 del(static_cast<T*>(item));
234 }
235
240 inline T* Get(const Extensible* container) const
241 {
242 return static_cast<T*>(GetRaw(container));
243 }
244
250 inline void Set(Extensible* container, T* value, bool sync = true)
251 {
252 if (container->extype != this->extype)
253 return;
254
255 T* old = static_cast<T*>(SetRaw(container, value));
256 Delete(container, old);
257 if (sync && synced)
258 Sync(container, value);
259 }
260
266 inline void Set(Extensible* container, const T& value, bool sync = true)
267 {
268 if (container->extype == this->extype)
269 Set(container, new T(value), sync);
270 }
271
276 template <typename... Args>
277 inline void SetFwd(Extensible* container, Args&&... args)
278 {
279 // Forwarded arguments are for complex types which are assumed to not
280 // be synced across the network. You can manually call Sync() if this
281 // is not the case.
282 if (container->extype == this->extype)
283 Set(container, new T(std::forward<Args>(args)...), false);
284 }
285
290 inline void Unset(Extensible* container, bool sync = true)
291 {
292 if (container->extype != this->extype)
293 return;
294
295 Delete(container, UnsetRaw(container));
296 if (synced && sync)
297 Sync(container, nullptr);
298 }
299};
