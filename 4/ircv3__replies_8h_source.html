50private:

55 ClientProtocol::EventProvider evprov;

61 void SendInternal(LocalUser* user, ClientProtocol::Message& msg)

63 ClientProtocol::Event ev(evprov, msg);

67 void SendNoticeInternal(LocalUser* user, Command* command, const std::string& description)
68 {
69 if (command)
70 user->WriteNotice(InspIRCd::Format("*** %s: %s", command->name.c_str(), description.c_str()));
71 else
72 user->WriteNotice(InspIRCd::Format("*** %s", description.c_str()));
73 }
74
75protected:
80 Reply(Module* Creator, const std::string& Cmd)
81 : cmd(Cmd)
82 , evprov(Creator, Cmd)
83 {
84 }
85
86public:
94 template<typename... Args>
95 void Send(LocalUser* user, Command* command, const std::string& code, Args&&... args)
96 {
97 static_assert(sizeof...(Args) >= 1);
98
99 ClientProtocol::Message msg(cmd.c_str(), ServerInstance->Config->GetServerName());
100 if (command)
101 msg.PushParamRef(command->name);
102 else
103 msg.PushParam("*");
104 msg.PushParam(code);
105 msg.PushParam(std::forward<Args>(args)...);
106 SendInternal(user, msg);
107 }
108
118 template<typename... Args>
119 void SendIfCap(LocalUser* user, const Cap::Capability& cap, Command* command, const std::string& code,
120 Args&&... args)
121 {
122 static_assert(sizeof...(Args) >= 1);
123
124 if (cap.IsEnabled(user))
125 Send(user, command, code, std::forward<Args>(args)...);
126 else
127 SendNoticeInternal(user, command, std::get<sizeof...(Args) - 1>(std::forward_as_tuple(args...)));
128 }
129};
