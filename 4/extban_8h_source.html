317inline bool ExtBan::Parse(const std::string& banentry, std::string& name, std::string& value, bool& inverted)
318{
319 // The mask must be in the format [!]<letter>:<value> or [!]<name>:<value>.
320 inverted = false;
321 size_t startpos = 0;
322 if (banentry[0] == '!')
323 {
324 inverted = true;
325 startpos++;
326 }
327
328 size_t endpos = banentry.find_first_not_of("ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz", startpos);
329 if (endpos == std::string::npos || banentry[endpos] != ':')
330 return false;
331
332 name.assign(banentry, startpos, endpos - startpos);
333 value.assign(banentry, endpos + 1);
334 return true;
335}
