50struct LDAPAttributes final
51 : public std::map<std::string, std::vector<std::string>>
52{
53 size_t size(const std::string& attr) const
54 {
55 const std::vector<std::string>& array = this->getArray(attr);
56 return array.size();
57 }
58
59 const std::vector<std::string> keys() const
60 {
61 std::vector<std::string> k;
62 for (const_iterator it = this->begin(), it_end = this->end(); it != it_end; ++it)
63 k.push_back(it->first);
64 return k;
65 }
66
67 const std::string& get(const std::string& attr) const
68 {
69 const std::vector<std::string>& array = this->getArray(attr);
70 if (array.empty())
71 throw LDAPException("Empty attribute " + attr + " in LDAPResult::get");
72 return array[0];
73 }
74
75 const std::vector<std::string>& getArray(const std::string& attr) const
76 {
77 const_iterator it = this->find(attr);
78 if (it == this->end())
79 throw LDAPException("Unknown attribute " + attr + " in LDAPResult::getArray");
80 return it->second;
81 }
