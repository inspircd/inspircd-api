231class ClientProtocol::Messages::Mode
232 : public ClientProtocol::Message
233{
234 Channel* chantarget;

245 static std::string ToModeLetters(const Modes::ChangeList::List& list, std::string::size_type maxlinelen, Modes::ChangeList::List::const_iterator beginit, Modes::ChangeList::List::const_iterator& lastit)
246 {
247 std::string ret;
248 std::string::size_type paramlength = 0;
249 char output_pm = '\0'; // current output state, '+' or '-'
250
251 Modes::ChangeList::List::const_iterator i;
252 for (i = beginit; i != list.end(); ++i)
253 {
254 const Modes::Change& item = *i;
255
256 const char needed_pm = (item.adding ? '+' : '-');
257 if (needed_pm != output_pm)
258 {
259 output_pm = needed_pm;
260 ret.push_back(output_pm);
261 }
262
263 if (!item.param.empty())
264 paramlength += item.param.length() + 1;
265 if (ret.length() + 1 + paramlength > maxlinelen)
266 {
267 // Mode sequence is getting too long
268 const char c = *ret.rbegin();
269 if ((c == '+') || (c == '-'))
270 ret.erase(ret.size()-1);
271 break;
272 }
273
274 ret.push_back(item.mh->GetModeChar());
275 }
276
277 lastit = i;
278 return ret;
279 }
280
283 void PushModeParams()
284 {
285 for (Modes::ChangeList::List::const_iterator i = beginit; i != lastit; ++i)
286 {
287 const Modes::Change& item = *i;
288 if (!item.param.empty())
289 PushParamRef(item.param);
290 }
291 }
292
293public:
298 static std::string ToModeLetters(const Modes::ChangeList& changelist)
299 {
300 // TODO: This assumes that std::string::max_size() >= UINT_MAX
301 Modes::ChangeList::List::const_iterator dummy;
302 return ToModeLetters(changelist.getlist(), UINT_MAX, changelist.getlist().begin(), dummy);
303 }
