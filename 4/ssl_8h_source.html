182 static void FlattenSendQueue(StreamSocket::SendQueue& sendq, size_t targetsize)
183 {
184 if ((sendq.size() <= 1) || (sendq.front().length() >= targetsize))
185 return;
186
187 // Avoid multiple repeated TLS encryption invocations
188 // This adds a single copy of the queue, but avoids
189 // much more overhead in terms of system calls invoked
190 // by an IOHook.
191 std::string tmp;
192 tmp.reserve(std::min(targetsize, sendq.bytes())+1);
193 do
194 {
195 tmp.append(sendq.front());
196 sendq.pop_front();
197 }
198 while (!sendq.empty() && tmp.length() < targetsize);
199 sendq.push_front(tmp);
200 }
201
202public:
203 static SSLIOHook* IsSSL(StreamSocket* sock)
204 {
205 IOHook* const lasthook = sock->GetLastHook();
206 if (lasthook && (lasthook->prov->type == IOHookProvider::IOH_SSL))
207 return static_cast<SSLIOHook*>(lasthook);
208
209 return NULL;
210 }
211
212 SSLIOHook(std::shared_ptr<IOHookProvider> hookprov)
213 : IOHook(hookprov)
214 {
215 }
216
221 virtual ssl_cert* GetCertificate() const
222 {
223 return certificate;
224 }
225
231 virtual std::string GetFingerprint() const
232 {
233 ssl_cert* cert = GetCertificate();
234 if (cert && cert->IsUsable())
235 return cert->GetFingerprint();
236 return "";
237 }

243 virtual void GetCiphersuite(std::string& out) const = 0;
244
249 virtual bool GetServerName(std::string& out) const = 0;
250
252 bool IsHookReady() const override { return status == STATUS_OPEN; }
