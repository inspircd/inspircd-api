187inline Regex::PatternPtr Regex::Engine::CreateHuman(const std::string& pattern) const
188{
189 if (pattern.empty() || pattern[0] != '/')
190 return Create(pattern, OPT_NONE);
191
192 size_t end = pattern.find_last_not_of("Ii");
193 if (!end || end == std::string::npos || pattern[end] != '/')
194 throw Exception(creator, pattern, "Regex patterns must be terminated with a '/'!");
195
196 uint8_t options = Regex::OPT_NONE;
197 for (const auto& flag : insp::iterator_range(pattern.begin() + end + 1, pattern.end()))
198 {
199 switch (flag)
200 {
201 case 'I':
202 options &= ~OPT_CASE_INSENSITIVE;
203 break;
204
205 case 'i':
206 options |= OPT_CASE_INSENSITIVE;
207 break;
208 }
209 }
210 return Create(pattern.substr(1, end - 1), options);
211}
