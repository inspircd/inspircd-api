88public:

91 {
92 push(chan->name);
93
94 const PrefixMode* pm = ServerInstance->Modes.FindNearestPrefixMode(rank);
95 if (pm)
96 push(InspIRCd::Format("You must be a channel %s or higher to %s.", pm->name.c_str(), message.c_str()));
97 else
98 push(InspIRCd::Format("You do not have the required channel privileges to %s.", message.c_str()));
99 }

102/* Helper for the ERR_INVALIDMODEPARAM numeric. */
103class Numerics::InvalidModeParameter final
104 : public Numeric::Numeric
105{
106private:
107 void push_message(ModeHandler* mode, const std::string& message)
108 {
109 if (!message.empty())
110 {
111 // The caller has specified their own message.
112 push(message);
113 return;
114 }
115
116 const std::string& syntax = mode->GetSyntax();
117 if (!syntax.empty())
118 {
119 // If the mode has a syntax hint we include it in the message.
120 push(InspIRCd::Format("Invalid %s mode parameter. Syntax: %s.", mode->name.c_str(), syntax.c_str()));
121 }
122 else
123 {
124 // Otherwise, send it without.
125 push(InspIRCd::Format("Invalid %s mode parameter.", mode->name.c_str()));
126 }
127 }
128
129public:
130 InvalidModeParameter(Channel* chan, ModeHandler* mode, const std::string& parameter, const std::string& message = "")
131 : Numeric(ERR_INVALIDMODEPARAM)
132 {
133 push(chan->name);
134 push(mode->GetModeChar());
135 push(parameter);
136 push_message(mode, message);
137 }
138
139 InvalidModeParameter(User* user, ModeHandler* mode, const std::string& parameter, const std::string& message = "")
140 : Numeric(ERR_INVALIDMODEPARAM)
141 {
142 push(user->registered & REG_NICK ? user->nick : "*");
143 push(mode->GetModeChar());
144 push(parameter);
145 push_message(mode, message);
146 }
147};
