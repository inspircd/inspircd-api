110  if (val.empty())
111  return def;
112

114  if (stdalgo::string::equalsci(val, enumval.first))
115  return enumval.second;
116 
117  std::vector<const char*> enumkeys;
118  std::transform(enumvals.begin(), enumvals.end(), std::back_inserter(enumkeys), [](const std::pair<const char*, TReturn>& ev) { return ev.first; });
119  throw CoreException(val + " is an invalid value for <" + name + ":" + key + ">; acceptable values are " +
120  stdalgo::string::join(enumkeys, ' ') + ", at " + source.str());
121  }
