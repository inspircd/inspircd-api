566inline bool SocketEngine::IgnoreError()
567{
568 if ((errno == EAGAIN) || (errno == EWOULDBLOCK))
569 return true;
570
571#ifdef _WIN32
572 if (WSAGetLastError() == WSAEWOULDBLOCK)
573 return true;
574#endif
575
576 return false;
