188inline bool Events::ModuleEventProvider::ElementComp::operator()(Events::ModuleEventListener* lhs, Events::ModuleEventListener* rhs) const
189{

197template<typename Class, typename... FunArgs, typename... FwdArgs>
198inline void Events::ModuleEventProvider::Call(void (Class::*function)(FunArgs...), FwdArgs&&... args) const
199{
200 if (GetModule() && GetModule()->dying)
201 return;
202
203 for (const auto& subscriber : GetSubscribers())
204 {
205 const Module* mod = subscriber->GetModule();
206 if (!mod || mod->dying)
207 continue;
208
209 Class* klass = static_cast<Class*>(subscriber);
210 (klass->*function)(std::forward<FwdArgs>(args)...);
211 }
212}

220 ModResult result;

222 {
223 const Module* mod = subscriber->GetModule();
224 if (!mod || mod->dying)
225 continue;
226
227 Class* klass = static_cast<Class*>(subscriber);
228 result = (klass->*function)(std::forward<FwdArgs>(args)...);
229 if (result != MOD_RES_PASSTHRU)
230 break;
231 }
232 return result;
